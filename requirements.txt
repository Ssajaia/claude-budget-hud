# -*- mode: python ; coding: utf-8 -*-
"""
PyInstaller build spec for Claude Budget HUD.

Usage:
    pyinstaller claude_budget_hud.spec

Outputs:
    dist/Claude Budget HUD          (macOS .app or Linux binary)
    dist/Claude Budget HUD.exe      (Windows)
"""

import sys
from pathlib import Path

ROOT = Path(".").resolve()

block_cipher = None

a = Analysis(
    [str(ROOT / "main.py")],
    pathex=[str(ROOT)],
    binaries=[],
    datas=[
        # Include any font files if present
        # (str(ROOT / "assets" / "fonts"), "assets/fonts"),
    ],
    hiddenimports=[
        # keyring backends
        "keyring.backends.Windows",
        "keyring.backends.macOS",
        "keyring.backends.SecretService",
        "keyring.backends.fail",
        # cryptography primitives pulled in at runtime
        "cryptography.hazmat.primitives.ciphers.aead",
        # PySide6 modules that may not be auto-detected
        "PySide6.QtSvg",
    ],
    hookspath=[],
    hooksconfig={},
    runtime_hooks=[],
    excludes=[
        "tkinter", "unittest", "email", "html", "http", "urllib",
        "xmlrpc", "doctest", "pdb", "difflib",
    ],
    win_no_prefer_redirects=False,
    win_private_assemblies=False,
    cipher=block_cipher,
    noarchive=False,
)

pyz = PYZ(a.pure, a.zipped_data, cipher=block_cipher)

# ── Platform-specific EXE / binary ────────────────────────────────────────────

exe = EXE(
    pyz,
    a.scripts,
    [],
    exclude_binaries=True,
    name="Claude Budget HUD",
    debug=False,
    bootloader_ignore_signals=False,
    strip=False,
    upx=True,
    console=False,          # no console window
    disable_windowed_traceback=False,
    target_arch=None,
    codesign_identity=None,
    entitlements_file=None,
    # Windows: embed icon
    icon=str(ROOT / "assets" / "icon.ico") if sys.platform == "win32" else None,
)

coll = COLLECT(
    exe,
    a.binaries,
    a.zipfiles,
    a.datas,
    strip=False,
    upx=True,
    upx_exclude=[],
    name="Claude Budget HUD",
)

# ── macOS .app bundle ─────────────────────────────────────────────────────────

if sys.platform == "darwin":
    app = BUNDLE(
        coll,
        name="Claude Budget HUD.app",
        icon=str(ROOT / "assets" / "icon.icns"),
        bundle_identifier="com.ssajaia.claude-budget-hud",
        info_plist={
            "CFBundleShortVersionString": "1.0.0",
            "CFBundleVersion": "1.0.0",
            "NSHighResolutionCapable": True,
            "LSUIElement": True,           # hide from Dock (runs as agent)
            "NSRequiresAquaSystemAppearance": False,
        },
    )
